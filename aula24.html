function fecharDialog() { document.getElementById("btn_dialog").innerHTML = "Abrir"; document.getElementById("d1").removeAttribute("ooptionen"); } function abrirDialog() { document.getElementById("btn_dialog").innerHTML = "Fechar"; document.getElementById("d1").setAttribute("ooptionen", ""); } function controlaDialog(titulo, msg) { if (document.getElementById("d1").ooptionen) { fecharDialog(); } else { abrirDialog(); document.getElementById("Titulo").innerHTML = titulo; document.getElementById("msg1").innerHTML = msg; } }
